param(
    [string]$DomainUser,
    [string]$DomainPass,
    [string]$Domain = "lab.local",
    [string]$BackendUrl = "",
    [string]$BackendToken = "",
    [string]$TargetHost = "",
    [string]$RequestedBy = "",
    [string]$RunId = "",
    [string]$ArtifactDir = "C:\Tools\PowerView\artifacts"
)

$ErrorActionPreference = "Stop"

$pvPath = "C:\Tools\PowerView\PowerView.ps1"

function Write-Info($msg) {
    Write-Host "[INFO] $msg"
}

function Write-WarnMsg($msg) {
    Write-Host "[WARN] $msg"
}

function Write-ErrMsg($msg) {
    Write-Host "[ERROR] $msg"
}

function Safe-Count($value) {
    if ($null -eq $value) { return 0 }
    if ($value -is [System.Array]) { return $value.Count }
    if ($value -is [System.Collections.IEnumerable] -and -not ($value -is [string])) {
        return @($value).Count
    }
    return 1
}

function Safe-Exec {
    param(
        [string]$Name,
        [scriptblock]$Script
    )
    try {
        $result = & $Script
        return @{
            ok = $true
            data = @($result)
            error = $null
        }
    }
    catch {
        return @{
            ok = $false
            data = @()
            error = $_.Exception.Message
        }
    }
}

function Select-SafeObject {
    param(
        [object[]]$InputObjects,
        [string[]]$Properties
    )

    $output = @()

    foreach ($obj in @($InputObjects)) {
        $row = [ordered]@{}
        foreach ($prop in $Properties) {
            try {
                $row[$prop] = $obj.$prop
            }
            catch {
                $row[$prop] = $null
            }
        }
        $output += [pscustomobject]$row
    }

    return $output
}

function Add-Recommendation {
    param(
        [System.Collections.Generic.List[string]]$List,
        [string]$Message
    )
    if (-not [string]::IsNullOrWhiteSpace($Message) -and -not $List.Contains($Message)) {
        $List.Add($Message)
    }
}

Write-Info "pv_collect_recon.ps1 started"
Write-Info "whoami: $(whoami)"
Write-Info "hostname: $env:COMPUTERNAME"
Write-Info "domain: $Domain"
Write-Info "domain user: $DomainUser"

if (-not (Test-Path $pvPath)) {
    Write-ErrMsg "PowerView.ps1 not found: $pvPath"
    exit 1
}

try {
    . $pvPath
    Write-Info "PowerView loaded successfully"
}
catch {
    Write-ErrMsg "Failed to load PowerView.ps1: $($_.Exception.Message)"
    exit 1
}

try {
    $sec = ConvertTo-SecureString $DomainPass -AsPlainText -Force
    $cred = New-Object System.Management.Automation.PSCredential($DomainUser, $sec)
}
catch {
    Write-ErrMsg "Failed to build PSCredential: $($_.Exception.Message)"
    exit 1
}

$collectedAt = (Get-Date).ToUniversalTime().ToString("o")
if (-not [string]::IsNullOrWhiteSpace($RunId)) {
    $runId = $RunId
}
else {
    $runId = "recon-" + (Get-Date -Format "yyyyMMddHHmmss")
}



Write-Info "ArtifactDir: $ArtifactDir"

try {
    New-Item -ItemType Directory -Force -Path $ArtifactDir | Out-Null
    Write-Info "ArtifactDir created or already exists"
}
catch {
    Write-ErrMsg "Failed to create ArtifactDir: $($_.Exception.Message)"
    exit 1
}

$runDir = Join-Path $ArtifactDir $runId
try {
    New-Item -ItemType Directory -Force -Path $runDir | Out-Null
    Write-Info "RunDir created: $runDir"
}
catch {
    Write-ErrMsg "Failed to create RunDir: $($_.Exception.Message)"
    exit 1
}



Write-Info "Collecting AD recon data..."

# 1) Users
$usersRaw = Safe-Exec -Name "users" -Script {
    Get-DomainUser -Domain $Domain -Credential $cred
}
$users = Select-SafeObject -InputObjects $usersRaw.data -Properties @(
    "samaccountname", "name", "distinguishedname", "memberof", "description"
)

# 2) Groups
$groupsRaw = Safe-Exec -Name "groups" -Script {
    Get-DomainGroup -Domain $Domain -Credential $cred
}
$groups = Select-SafeObject -InputObjects $groupsRaw.data -Properties @(
    "samaccountname", "name", "distinguishedname", "description"
)

# 3) Computers
$computersRaw = Safe-Exec -Name "computers" -Script {
    Get-DomainComputer -Domain $Domain -Credential $cred
}
$computers = Select-SafeObject -InputObjects $computersRaw.data -Properties @(
    "dnshostname", "name", "samaccountname", "operatingsystem", "distinguishedname"
)

# 4) Domain Admins
$domainAdminsRaw = Safe-Exec -Name "domain_admins" -Script {
    Get-DomainGroupMember -Identity "Domain Admins" -Domain $Domain -Credential $cred
}
$domainAdmins = Select-SafeObject -InputObjects $domainAdminsRaw.data -Properties @(
    "MemberName", "MemberDistinguishedName", "MemberObjectClass"
)

# 5) Enterprise Admins
$enterpriseAdminsRaw = Safe-Exec -Name "enterprise_admins" -Script {
    Get-DomainGroupMember -Identity "Enterprise Admins" -Domain $Domain -Credential $cred
}
$enterpriseAdmins = Select-SafeObject -InputObjects $enterpriseAdminsRaw.data -Properties @(
    "MemberName", "MemberDistinguishedName", "MemberObjectClass"
)

# 6) DnsAdmins
$dnsAdminsRaw = Safe-Exec -Name "dns_admins" -Script {
    Get-DomainGroupMember -Identity "DnsAdmins" -Domain $Domain -Credential $cred
}
$dnsAdmins = Select-SafeObject -InputObjects $dnsAdminsRaw.data -Properties @(
    "MemberName", "MemberDistinguishedName", "MemberObjectClass"
)

# 7) SPN users (Kerberoast candidates)
$spnUsersRaw = Safe-Exec -Name "spn_users" -Script {
    Get-DomainUser -SPN -Domain $Domain -Credential $cred
}
$spnUsers = Select-SafeObject -InputObjects $spnUsersRaw.data -Properties @(
    "samaccountname", "serviceprincipalname", "distinguishedname", "description"
)

# 8) No-preauth users (AS-REP candidates)
$noPreauthRaw = Safe-Exec -Name "no_preauth_users" -Script {
    Get-DomainUser -PreauthNotRequired -Domain $Domain -Credential $cred
}
$noPreauthUsers = Select-SafeObject -InputObjects $noPreauthRaw.data -Properties @(
    "samaccountname", "distinguishedname", "description"
)

# 9) OUs
$ousRaw = Safe-Exec -Name "ous" -Script {
    Get-DomainOU -Domain $Domain -Credential $cred
}
$ous = Select-SafeObject -InputObjects $ousRaw.data -Properties @(
    "name", "distinguishedname", "gplink"
)

# 10) Trusts
$trustsRaw = Safe-Exec -Name "trusts" -Script {
    Get-DomainTrust -Domain $Domain -Credential $cred
}
$trusts = Select-SafeObject -InputObjects $trustsRaw.data -Properties @(
    "SourceName", "TargetName", "TrustType", "TrustDirection"
)

# 11) Optional: interesting ACL path candidates
$aclInterestingRaw = Safe-Exec -Name "interesting_acls" -Script {
    Find-InterestingDomainAcl -Domain $Domain -Credential $cred
}
$interestingAcls = Select-SafeObject -InputObjects $aclInterestingRaw.data -Properties @(
    "ObjectDN", "AceQualifier", "ActiveDirectoryRights", "SecurityIdentifier"
)

$recommendations = New-Object System.Collections.Generic.List[string]

if ((Safe-Count $spnUsers) -gt 0) {
    Add-Recommendation -List $recommendations -Message "kerberoasting"
}
if ((Safe-Count $noPreauthUsers) -gt 0) {
    Add-Recommendation -List $recommendations -Message "asrep_roasting"
}
if ((Safe-Count $dnsAdmins) -gt 0) {
    Add-Recommendation -List $recommendations -Message "dns_admin_abuse"
}
if ((Safe-Count $domainAdmins) -gt 0) {
    Add-Recommendation -List $recommendations -Message "privilege_escalation_tracking"
}
if ((Safe-Count $interestingAcls) -gt 0) {
    Add-Recommendation -List $recommendations -Message "ad_dacl_abuse"
}

$errors = @{}
$collectorStates = @(
    @{ name = "users"; ok = $usersRaw.ok; error = $usersRaw.error },
    @{ name = "groups"; ok = $groupsRaw.ok; error = $groupsRaw.error },
    @{ name = "computers"; ok = $computersRaw.ok; error = $computersRaw.error },
    @{ name = "domain_admins"; ok = $domainAdminsRaw.ok; error = $domainAdminsRaw.error },
    @{ name = "enterprise_admins"; ok = $enterpriseAdminsRaw.ok; error = $enterpriseAdminsRaw.error },
    @{ name = "dns_admins"; ok = $dnsAdminsRaw.ok; error = $dnsAdminsRaw.error },
    @{ name = "spn_users"; ok = $spnUsersRaw.ok; error = $spnUsersRaw.error },
    @{ name = "no_preauth_users"; ok = $noPreauthRaw.ok; error = $noPreauthRaw.error },
    @{ name = "ous"; ok = $ousRaw.ok; error = $ousRaw.error },
    @{ name = "trusts"; ok = $trustsRaw.ok; error = $trustsRaw.error },
    @{ name = "interesting_acls"; ok = $aclInterestingRaw.ok; error = $aclInterestingRaw.error }
)

foreach ($state in $collectorStates) {
    if (-not $state.ok -and $state.error) {
        $errors[$state.name] = $state.error
    }
}

$summary = [ordered]@{
    total_users = Safe-Count $users
    total_groups = Safe-Count $groups
    total_computers = Safe-Count $computers
    domain_admins_count = Safe-Count $domainAdmins
    enterprise_admins_count = Safe-Count $enterpriseAdmins
    dns_admins_count = Safe-Count $dnsAdmins
    spn_users_count = Safe-Count $spnUsers
    no_preauth_users_count = Safe-Count $noPreauthUsers
    ous_count = Safe-Count $ous
    trusts_count = Safe-Count $trusts
    interesting_acls_count = Safe-Count $interestingAcls
}

$payload = [ordered]@{
    tool = "powerview"
    query_type = "recon_overview"
    collected_at = $collectedAt
    run_id = $runId
    target_host = $TargetHost
    requested_by = $RequestedBy
    executed_host = $env:COMPUTERNAME
    executed_user = (whoami)
    domain = $Domain
    summary = $summary
    recommendations = @($recommendations)
    errors = $errors
    data = [ordered]@{
        users = $users
        groups = $groups
        computers = $computers
        domain_admins = $domainAdmins
        enterprise_admins = $enterpriseAdmins
        dns_admins = $dnsAdmins
        spn_users = $spnUsers
        no_preauth_users = $noPreauthUsers
        ous = $ous
        trusts = $trusts
        interesting_acls = $interestingAcls
    }
}

$json = $payload | ConvertTo-Json -Depth 8

$jsonPath = Join-Path $runDir "powerview_recon.json"
$summaryPath = Join-Path $runDir "summary.json"


$utf8NoBom = New-Object System.Text.UTF8Encoding($false)

[System.IO.File]::WriteAllText($jsonPath, $json, $utf8NoBom)
[System.IO.File]::WriteAllText(
    $summaryPath,
    ($payload.summary | ConvertTo-Json -Depth 4),
    $utf8NoBom
)

# latest snapshot
$latestDir = Join-Path $ArtifactDir "latest"
New-Item -ItemType Directory -Force -Path $latestDir | Out-Null

[System.IO.File]::WriteAllText(
    (Join-Path $latestDir "powerview_recon.json"),
    $json,
    $utf8NoBom
)

[System.IO.File]::WriteAllText(
    (Join-Path $latestDir "summary.json"),
    ($payload.summary | ConvertTo-Json -Depth 4),
    $utf8NoBom
)

Write-Info "Recon JSON saved: $jsonPath"
Write-Info "Summary JSON saved: $summaryPath"
Write-Info "Summary => users=$($summary.total_users), groups=$($summary.total_groups), computers=$($summary.total_computers), spn=$($summary.spn_users_count), no_preauth=$($summary.no_preauth_users_count), dnsadmins=$($summary.dns_admins_count)"



# backend send (best-effort)
Write-Info "BackendUrl: $BackendUrl"

if (-not [string]::IsNullOrWhiteSpace($BackendUrl)) {
    try {
        $endpoint = ($BackendUrl.TrimEnd("/") + "/recon-results")

        $headers = @{}
        if (-not [string]::IsNullOrWhiteSpace($BackendToken)) {
            $headers["X-API-Token"] = $BackendToken
        }

        $jsonBytes = [System.Text.Encoding]::UTF8.GetBytes($json)

        Write-Info "Sending recon payload to backend: $endpoint"
        Write-Info "Payload size(bytes): $($jsonBytes.Length)"

        $response = Invoke-RestMethod `
            -Uri $endpoint `
            -Method Post `
            -Headers $headers `
            -ContentType "application/json; charset=utf-8" `
            -Body $jsonBytes `
            -TimeoutSec 60

        Write-Info "Backend send success: $($response | ConvertTo-Json -Compress)"
    }
    catch {
        Write-WarnMsg "Backend send failed: $($_.Exception.Message)"
    }
}
else {
    Write-WarnMsg "BackendUrl not provided, skip backend send"
}




# stdout에도 요약 출력
Write-Host "================ RECON SUMMARY ================"
Write-Host "Total Users           : $($summary.total_users)"
Write-Host "Total Groups          : $($summary.total_groups)"
Write-Host "Total Computers       : $($summary.total_computers)"
Write-Host "Domain Admins         : $($summary.domain_admins_count)"
Write-Host "Enterprise Admins     : $($summary.enterprise_admins_count)"
Write-Host "DnsAdmins             : $($summary.dns_admins_count)"
Write-Host "SPN Users             : $($summary.spn_users_count)"
Write-Host "NoPreAuth Users       : $($summary.no_preauth_users_count)"
Write-Host "Interesting ACLs      : $($summary.interesting_acls_count)"
Write-Host "==============================================="

if ($recommendations.Count -gt 0) {
    Write-Host "[INFO] Recommended next scenarios:"
    foreach ($r in $recommendations) {
        Write-Host " - $r"
    }
}

exit 0
